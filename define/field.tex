\newcommand{\field}[4]{
  \begin{tabular}{|p{3.5cm}|c|m{1cm}|m{2cm}|}
   \cline{1-1} \cline{3-4}
   #1 & \hspace{4cm} & #2 & #3 \\
   \hline
   \multicolumn{4}{|p{\textwidth}|}{#4} \\
   \hline
  \end{tabular}
}

\newcommand{\bitfield}[1]
{
    \begin{rightwordgroup}{Type (\texttt{#1})}
      \wordbox{1}{\centering \texttt{#1}}
    \end{rightwordgroup} \\
}
